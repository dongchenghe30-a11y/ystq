// Basic Usage
const visualizer = new TabVisualizer({
    container: document.getElementById('visualizer'),
    tabManager: myTabManager,
    layout: 'tree',
    theme: 'light'
});

// With custom callbacks
const visualizer = new TabVisualizer({
    container: document.getElementById('visualizer'),
    layout: 'waterfall',
    onClick: (tab) => {
        console.log('Tab clicked:', tab);
        // Custom navigation logic
    },
    onHover: (tab) => {
        console.log('Tab hovered:', tab);
        // Custom hover logic
    }
});

// Change layout
visualizer.setLayout('waterfall');

// Change theme
visualizer.setTheme('dark');

// Refresh data
visualizer.refresh();

// Destroy
visualizer.destroy();
